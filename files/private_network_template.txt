auto vmbr1
iface vmbr1 inet static
    address #PRIVATE_ADDRESS#
    bridge-ports none
    bridge-stp off
    bridge-fd 0
    post-up   iptables -t nat -A POSTROUTING -s '#PRIVATE_CIDR#' -o vmbr0 -j MASQUERADE
    post-down iptables -t nat -D POSTROUTING -s '#PRIVATE_CIDR#' -o vmbr0 -j MASQUERADE
    post-up   iptables -t raw -I PREROUTING -i fwbr+ -j CT --zone 1
    post-down iptables -t raw -D PREROUTING -i fwbr+ -j CT --zone 1
